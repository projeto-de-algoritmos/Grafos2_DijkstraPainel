# Exemplo
		Inserir os nós: 
		1. A, B, 10
		2. A, C, 15
		3. B, D, 12
		4. B, F, 15
		5. C, E, 10
		6. D, E, 2
		7. D, F, 1
		8. F, E, 5

		Resultado:

		Node: E
		Shortest Path: [Node A, Node B, Node D]
		Distance: 24

		Node: F
		Shortest Path: [Node A, Node B, Node D]
		Distance: 23

		Node: D
		Shortest Path: [Node A, Node B]
		Distance: 22

		Node: B
		Shortest Path: [Node A]
		Distance: 10

		Node: C
		Shortest Path: [Node A]
		Distance: 15

		Node: A
		Shortest Path: []
		Distance: 0
